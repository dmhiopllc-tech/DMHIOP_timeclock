async function saveEditEntry(event) { event.preventDefault(); const entryId = document.getElementById('edit-entry-id').value; const clockInValue = document.getElementById('edit-clock-in').value; const clockOutValue = document.getElementById('edit-clock-out').value; const reason = document.getElementById('edit-reason').value; if (reason.trim().length < 5) { alert('⚠️ Please provide a more detailed reason (at least 5 characters)'); return; } try { // 🔥 FIXED: Convert local datetime-local input to UTC // datetime-local gives us the time as displayed (already in MST from user input) // We need to parse it as MST and convert to UTC for storage const clockInParts = clockInValue.split(/[-T:]/); const clockInUTC = new Date(Date.UTC( parseInt(clockInParts[0]), parseInt(clockInParts[1]) - 1, parseInt(clockInParts[2]), parseInt(clockInParts[3]) + 7, // Add 7 hours to convert MST to UTC parseInt(clockInParts[4]) )); const clockInISO = clockInUTC.toISOString(); let clockOutISO = null; if (clockOutValue) { const clockOutParts = clockOutValue.split(/[-T:]/); const clockOutUTC = new Date(Date.UTC( parseInt(clockOutParts[0]), parseInt(clockOutParts[1]) - 1, parseInt(clockOutParts[2]), parseInt(clockOutParts[3]) + 7, // Add 7 hours to convert MST to UTC parseInt(clockOutParts[4]) )); clockOutISO = clockOutUTC.toISOString(); // Validate clock out is after clock in if (new Date(clockOutISO) <= new Date(clockInISO)) { alert('⚠️ Clock out time must be after clock in time'); return; } } // Get the original entry for preserving adjustment reason const entry = currentTimeEntries.find(e => e.id === entryId); const updates = { clock_in_time: clockInISO, clock_out_time: clockOutISO, manually_adjusted: true, adjusted_by: currentUser.id, adjusted_at: new Date().toISOString(), adjustment_reason: entry?.adjustment_reason ? `${entry.adjustment_reason}\n\n[EDITED] ${reason}` : `[EDITED] ${reason}` }; const { error } = await db .from('time_entries') .update(updates) .eq('id', entryId); if (error) throw error; const hours = clockOutISO ? calculateHours(clockInISO, clockOutISO) : 0; alert( `✅ TIME ENTRY UPDATED!\n\n` + `Clock In: ${formatTime(clockInISO)} MST\n` + `Clock Out: ${clockOutISO ? formatTime(clockOutISO) + ' MST' : 'Still clocked in'}\n` + `Hours: ${hours.toFixed(2)}` ); closeEditEntryModal(); await loadTimeData(); } catch (error) { console.error('Error updating entry:', error); alert(`❌ ERROR: Failed to update entry\n\n${error.message}`); } }
